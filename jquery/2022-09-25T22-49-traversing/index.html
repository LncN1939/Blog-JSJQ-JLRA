jQuery Traversing

Publicado el 25 de septiembre de 2022

por Armando Meza

Los métodos de traversing de jQuery sirven para recorrer el DOM de distintas maneras, seleccionar y encontrar elementos específicos.

Métodos Traversing de jQuery

add()

Añade elementos al conjunto de elementos seleccionados

$("div").add("h1").add("p");
// Resultado
// <div>
//   <h1></h1>
//   <p></p>
// </div>

children()

Retorna todos los hijos directos del elemento seleccionado

$("div").children();
// Resultado 
// {
//   0: <h1></h1>,
//   1: <p></p>,
//   ...
// }

closest()

Retorna el primer ancestro del elemento seleccionado, basado en la selección. 


// <div>
//   <h1></h1>
//   <p></p>
// </div>
$("h1").closest("div");
//Resultado
// {
//   0: <div></div>,
//   ...
// }

contents()

Retorna todos los hijos directos del elemento seleccionado, incluyendo el contenido de texto y comentarios 

// <div>
//   <h1></h1>
//   <p></p>
// </div>

$("div").contents();

// Resultado
// {
//   0: {},
//   1: <h1></h1>,
//   2: {},
//   3: <p></p>,
//   4: {},
// }

each()

Ejecuta una función por cada elemento seleccionado.

// <div>
//   <h1></h1>
//   <p></p>
// </div>

$("div").children().each(function(i, e) {console.log(e)})

// Resultado
// <h1></h1>
// <p></p>

eq()

Retorna un elemento con el índice especificado.

// <div>
//   <h1></h1>
//   <p></p>
// </div>
$("div").children().eq(0);
// Resultado
// <h1></h1>

filter()

Reduce el conjunto de elementos seleccionados a aquellos que cumplen cierta condición. 

// <div>
//   <h1></h1>
//   <p></p>
// </div>
$("div").children().filter("p");
// Resultado
// <p></p>

find()

Retorna los elementos descendientes del elemento seleccionado

// <div>
//   <h1></h1>
//   <p></p>
// </div>
$("div").find("p");
// Resultado
// <p></p>

first()

Retorna el primer elemento de los seleccionados

// <div>
//   <h1></h1>
//   <p></p>
// </div>
$("div").children().first();
// Resultado
// <h1></h1>

has()

Retorna todos los elementos que tienen uno o más elementos dentro de ellos. 

// <div>
//   <h1></h1>
//   <p></p>
// </div>
$("div").has("h1");
// Resultado
// <div>
//   <h1></h1>
//   <p></p>
// </div>

is()

Chequea los elementos seleccionados por un selector, y retorna true si al menos uno de estos cumple con los argumentos dados 

$("div").is("div");
// true
$("div").is("h1");
// false

last()

Retorna el último elemento de los seleccionados.

// <div>
//   <h1></h1>
//   <p></p>
// </div>
$("div").last();
// Resultado
// <p></p>

map()

Retorna un nuevo objeto jQuery conteniendo los elementos que retorna la función pasada como argumento. 

// <div>
//   <h1>Hola</h1>
//   <p>Buenas</p>
// </div>
$("div").children().map(function(i, e) {return $(e).text()})
// Resultado
// {
//   0: "Hola",
//   1: "Buenas"
// }

next()

Retorna el siguiente elemento hermano del elemento seleccionado.

// <div>
//   <h1></h1>
//   <p></p>
// </div>
$("h1").first().next();
// Resultado
// <p></p>

nextAll()

Retorna todos los elementos hermanos siguientes del elemento seleccionado. 

// <div>
//   <h1></h1>
//   <p></p>
//   <p></p>
//   <p></p>
// </div>
$("h1").first().nextAll();
// Resultado
// {
//   0: <p></p>,
//   1: <p></p>,
//   2: <p></p>,
// }

nextUntil()

Retorna todos los elementos hermanos siguientes entre dos argumentos dados. 

// <div>
//   <h1></h1>
//   <p></p>
//   <p></p>
//   <p class="stop"></p>
//   <p></p>
// </div>
$("h1").first().nextUntil("p.stop");
// Resultado
// {
//   0: <p></p>,
//   1: <p></p>,
// }

not()

Retorna los elementos que no cumplen cierta condición.

// <div>
//   <h1></h1>
//   <p></p>
//   <p></p>
//   <h2></h2>
// </div>
$("div").children().not("p");
// Resultado
// {
//   0: <h1></h1>,
//   1: <h2></h2>,
//   ...
// }

parent()

Retorna el primer elemento padre directo del elemento seleccionado. 

// <div class="grandparent">
//   <div class="parent">
//     <h1></h1>
//     <p></p>
//   </div>
// </div> 
$("h1").parent();
// Resultado
// <div class="parent"></div>

parents()

Retorna todos los ancestros del elemento seleccionado.

// <div class="grandparent">
//   <div class="parent">
//     <h1></h1>
//     <p></p>
//   </div>
// </div>
$("h1").parents();
// Resultado
// {
//   0: <div class="parent"></div>,
//   1: <div class="grandparent"></div>
// }

parentsUntil()

Retorna todos los ancestros entre dos argumentos dados.

// <div class="stop">
//   <div class="grandparent">
//     <div class="parent">
//       <h1></h1>
//       <p></p>
//     </div>
//   </div>
// </div>
$("h1").parentsUntil("div.stop");
// Resultado
// {
//   0: <div class="parent"></div>,
//   1: <div class="grandparent"></div>
// }

prev()

Retorna todos los elementos hermanos anteriores del elemento seleccionado 

// <div>
//   <h1></h1>
//   <p></p>
// </div>
$("p").first().prev();
// Resultado
// <h1></h1>

prevAll()

Retorna todos los elementos hermanos anteriores del elemento seleccionado. 

// <div>
//   <h1></h1>
//   <p></p>
//   <p></p>
//   <h2></h2>
// </div>
$("h2").first().prevAll();
// Resultado
// {
//   0: <p></p>,
//   1: <p></p>,
//   2: <h1></h1>
// }

prevUntil()

Retorna todos los elementos hermanos anteriores entre dos argumentos dados. 

// <div>
//   <h1 class="stop"></h1>
//   <p></p>
//   <p></p>
//   <h2></h2>
// </div>
$("h2").first().prevUntil(".stop");
// Resultado
// {
//   0: <p></p>,
//   1: <p></p>,
// }

siblings()

Retorna todos los elementos hermanos del elemento seleccionado.

// <div>
//   <h1 class="stop"></h1>
//   <p></p>
//   <p></p>
//   <h2></h2>
// </div>
$("h1").first().siblings();
// Resultado
// {
//   0: <p></p>,
//   1: <p></p>,
//   2: <h2></h2>
// }

slice()

Reduce el conjunto de elementos seleccionados a un conjunto especificado por un rango de índices 

// <div>
//   <h1 class="stop"></h1>
//   <p></p>
//   <p></p>
//   <h2></h2>
// </div>
$("div").children().slice(2, 4);    
// Resultado
// {
//   0: <p></p>,
//   1: <h2></h2>
// }
